# 循环使用

## for

### 数字循环

```lua
-- 0-5 每次步长为1
for var=0,3,1 do  
   print(var)  
end
--0  
--1  
--2  
--3  

for i=3,1,-1 do  
    print(i)  
end
--3
--2
--1

-- 使用函数返回形式
function maxAndStep()  
    return 3,1  
end  
  
for i=1,maxAndStep() do  
    print(i)  
end

function maxAndStep()  
    return {1,3,1}  
end  
  
for i=maxAndStep()[1],maxAndStep()[2],maxAndStep()[3] do  
    print(i)  
end



```

### 迭代

```lua
a = {"one", "two", "three"}  
for i, v in ipairs(a) do  
    print(i, v)  
end  

1 one  
2 two  
3 three
```

## while

```lua
a=10  
while( a < 14 )  
do  
   print("a 的值为:", a)  
   a = a+1  
end

a 的值为:    10
a 的值为:    11
a 的值为:    12
a 的值为:    13
```


## repeat 

```lua
--[ 变量定义 --]  
a = 10  
--[ 执行循环 --]  
repeat  
   print("a的值为:", a)  
   a = a + 1  
until( a > 13 )


a的值为:    10
a的值为:    11
a的值为:    12
a的值为:    13

```
## goto

``` lua
i = 10  
while i > 0 do  
    i = i-1;  
    print(i);  
    if i ==5 then  
        goto exception;  
    end  
end  
  
::exception::  
print("error!!!");

--9  
--8  
--7  
--6  
--5  
--error!!!
```
